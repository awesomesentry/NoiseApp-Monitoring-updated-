-- WARNING: This schema is for context only and is not meant to be run.
-- Table order and constraints may not be valid for execution.

CREATE TABLE public.sitters (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT sitters_pkey PRIMARY KEY (id)
);
CREATE TABLE public.profiles (
  id uuid NOT NULL,
  role text NOT NULL DEFAULT 'teacher'::text CHECK (role = ANY (ARRAY['admin'::text, 'teacher'::text])),
  full_name text,
  mobile text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT profiles_pkey PRIMARY KEY (id),
  CONSTRAINT profiles_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
CREATE TABLE public.classrooms (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  name text NOT NULL UNIQUE,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT classrooms_pkey PRIMARY KEY (id)
);
CREATE TABLE public.teacher_classrooms (
  teacher_id uuid NOT NULL,
  classroom_id uuid NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT teacher_classrooms_pkey PRIMARY KEY (teacher_id, classroom_id),
  CONSTRAINT teacher_classrooms_teacher_id_fkey FOREIGN KEY (teacher_id) REFERENCES public.profiles(id),
  CONSTRAINT teacher_classrooms_classroom_id_fkey FOREIGN KEY (classroom_id) REFERENCES public.classrooms(id)
);
CREATE TABLE public.noise_events (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  device_id text NOT NULL,
  warning_level text NOT NULL,
  warning_color text NOT NULL,
  duration_seconds integer NOT NULL,
  decibel numeric NOT NULL,
  audio_url text,
  level text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  room text,
  subject text,
  teacher_name text,
  buzzer_triggered boolean NOT NULL DEFAULT false,
  audio_recorded boolean NOT NULL DEFAULT false,
  classroom_id uuid,
  event_group_id text,
  event_ts_ms bigint,
  event_time_utc timestamp with time zone DEFAULT to_timestamp((((event_ts_ms)::numeric / 1000.0))::double precision),
  CONSTRAINT noise_events_pkey PRIMARY KEY (id),
  CONSTRAINT noise_events_classroom_id_fkey FOREIGN KEY (classroom_id) REFERENCES public.classrooms(id)
);
CREATE TABLE public.noise_event_audio (
  noise_event_id uuid NOT NULL,
  audio_url text NOT NULL,
  audio_seconds integer NOT NULL DEFAULT 5,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT noise_event_audio_pkey PRIMARY KEY (noise_event_id),
  CONSTRAINT noise_event_audio_noise_event_id_fkey FOREIGN KEY (noise_event_id) REFERENCES public.noise_events(id)
);
CREATE TABLE public.audio_reviews (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  noise_event_id uuid NOT NULL,
  reviewer_id uuid NOT NULL,
  review_text text NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT audio_reviews_pkey PRIMARY KEY (id),
  CONSTRAINT audio_reviews_noise_event_id_fkey FOREIGN KEY (noise_event_id) REFERENCES public.noise_events(id),
  CONSTRAINT audio_reviews_reviewer_id_fkey FOREIGN KEY (reviewer_id) REFERENCES public.profiles(id)
);
CREATE TABLE public.audit_logs (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  actor_id uuid,
  action text NOT NULL,
  detail text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT audit_logs_pkey PRIMARY KEY (id),
  CONSTRAINT audit_logs_actor_id_fkey FOREIGN KEY (actor_id) REFERENCES public.profiles(id)
);
CREATE TABLE public.teacher_schedules (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  teacher_id uuid NOT NULL,
  day text NOT NULL CHECK (day = ANY (ARRAY['Monday'::text, 'Tuesday'::text, 'Wednesday'::text, 'Thursday'::text, 'Friday'::text, 'Saturday'::text, 'Sunday'::text])),
  start_time time without time zone NOT NULL,
  end_time time without time zone NOT NULL,
  subject text,
  room text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT teacher_schedules_pkey PRIMARY KEY (id),
  CONSTRAINT teacher_schedules_teacher_id_fkey FOREIGN KEY (teacher_id) REFERENCES public.profiles(id)
);
CREATE TABLE public.noise_db_series (
  id bigint NOT NULL DEFAULT nextval('noise_db_series_id_seq'::regclass),
  device_id text NOT NULL,
  ts_ms bigint NOT NULL,
  db10 integer NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  ts timestamp with time zone DEFAULT to_timestamp((((ts_ms)::numeric / 1000.0))::double precision),
  CONSTRAINT noise_db_series_pkey PRIMARY KEY (id)
);
